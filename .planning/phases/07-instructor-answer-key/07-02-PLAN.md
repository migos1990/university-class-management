// GET /sca/findings/:id -- Detail view (shared by students and instructors)
router.get('/findings/:id', requireAuth, (req, res) => {
  const finding = db.prepare('SELECT * FROM sca_findings WHERE id = ?').get(parseInt(req.params.id));
  if (!finding) return res.status(404).render('error', { ... });

  const user = req.session.user;
  let myReview = null;
  let allReviews = [];

  if (user.role === 'student') {
    myReview = db.prepare('SELECT * FROM sca_student_reviews WHERE finding_id = ? AND student_id = ?').get(finding.id, user.id);
  } else {
    allReviews = db.prepare('SELECT * FROM sca_student_reviews WHERE finding_id = ?').all(finding.id);
  }

  const vmEntry = db.prepare('SELECT * FROM vulnerabilities WHERE source = ? AND source_id = ?').get('sca', finding.id);
  const lang = req.session.language || 'fr';
  const localizedFinding = localize(finding, lang);
  localizedFinding.difficulty = DIFFICULTY_MAP[finding.id] || 'medium';
  const users = user.role !== 'student' ? db.prepare('SELECT id, username FROM users WHERE role = ?').all('student') : null;

  res.render('sca/finding-detail', {
    title: localizedFinding.title,
    finding: localizedFinding,
    myReview,
    allReviews,
    vmEntry,
    users,
    needsPrism: true
  });
});

**Step B -- Modify template (views/sca/finding-detail.ejs):**

Add a collapsible answer section in the LEFT column, AFTER the remediation paragraph closing `</div>` (end of the finding info card, line 78) and BEFORE the student reviews section (line 80).

The section must be wrapped in an EJS conditional that PREVENTS HTML emission for students:

```ejs
<%# AKEY-05: Inline answer -- NOT rendered for students (absent from page source) %>
<% if (user.role !== 'student' && answerKey) { %>
<div class="card" style="margin-bottom:1rem; border-left:4px solid #002855;">
  <details>
    <summary style="cursor:pointer; font-weight:600; color:#002855; padding:0.5rem 0;">
      <%= t('sca.answerKey.inlineTitle') %>
    </summary>
    <div style="margin-top:0.75rem;">
      <p style="margin-bottom:0.5rem;">
        <strong><%= t('sca.answerKey.expectedClassification') %> :</strong>
        <span class="badge-sm <%= answerKey.classification.includes('investigation') ? 'cls-needs' : 'cls-confirmed' %>">
          <%= answerKey.classification %>
        </span>
      </p>
      <p style="margin-bottom:0.5rem; line-height:1.7;">
        <strong><%= t('sca.answerKey.reasoning') %> :</strong>
        <%= answerKey.reasoning %>
      </p>
      <p style="line-height:1.7;">
        <strong><%= t('sca.answerKey.discussionPrompt') %> :</strong>
        <em><%= answerKey.discussion %></em>
      </p>
    </div>
  </details>
</div>
<% } %>
```

CRITICAL: Use `<% if %>` (server-side conditional), NOT `style="display:none"`. The `<% if %>` block prevents the HTML from being generated in the response for students. This means students cannot see the answer even by viewing page source. The double protection is: (1) `answerKey` is `null` for students from the route handler, and (2) the EJS conditional checks both `user.role !== 'student'` AND `answerKey` is truthy.

Use `<%= %>` (escaped output) for all answer key text to prevent XSS.

The `cls-needs` vs `cls-confirmed` badge class is determined by checking if the classification string contains "investigation" -- finding 11 uses "Necessite une investigation" while all others use "Vrai positif (confirme)".

Add the following test section using the existing `profCookie` from Phase D and `studentCookie` from Phase C:

```javascript
  // -----------------------------------------------------------------------
  // Phase E: Answer Key Role-Gating (AKEY-01, AKEY-04)
  // -----------------------------------------------------------------------
  console.log('--- Phase E: Answer Key Role-Gating ---');
  console.log('');

  // Test 1: Professor can access answer key
  if (profCookie) {
    try {
      const akRes = await request({
        url: `http://localhost:${DASHBOARD_PORT}/sca/answer-key`,
        headers: { 'Cookie': profCookie }
      });
      if (akRes.statusCode === 200 && akRes.body.includes('Corrig')) {
        console.log('  ✓ Answer key -- professor access OK');
      } else {
        console.log(`  ✗ Answer key -- professor got status ${akRes.statusCode}, missing French content`);
        allPassed = false;
      }
    } catch (e) {
      console.log(`  ✗ Answer key -- professor access error: ${e.message}`);
      allPassed = false;
    }
  }

  // Test 2: Student is denied answer key access
  if (studentCookie) {
    try {
      const akStudentRes = await request({
        url: `http://localhost:${DEEP_PORT}/sca/answer-key`,
        headers: { 'Cookie': studentCookie }
      });
      if (akStudentRes.statusCode === 403) {
        console.log('  ✓ Answer key -- student denied (403)');
      } else {
        console.log(`  ✗ Answer key -- student got status ${akStudentRes.statusCode} (expected 403)`);
        allPassed = false;
      }
    } catch (e) {
      console.log(`  ✗ Answer key -- student denial check error: ${e.message}`);
      allPassed = false;
    }
  }

  // Test 3: Student page source does NOT contain answer key data
  if (studentCookie) {
    try {
      const findingRes = await request({
        url: `http://localhost:${DEEP_PORT}/sca/findings/1`,
        headers: { 'Cookie': studentCookie }
      });
      if (findingRes.statusCode === 200 && !findingRes.body.includes('answerKey') && !findingRes.body.includes('sca.answerKey.inlineTitle')) {
        console.log('  ✓ Finding detail -- no answer key in student page source');
      } else {
        console.log('  ✗ Finding detail -- answer key content leaked to student page source!');
        allPassed = false;
      }
    } catch (e) {
      console.log(`  ✗ Finding detail -- student source check error: ${e.message}`);
      allPassed = false;
    }
  }

  console.log('');
```
