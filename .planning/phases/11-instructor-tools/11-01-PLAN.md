  **Step 3 -- Add client-side JavaScript** in the existing `<script>` block, BEFORE the `refreshStats()` function definition. Add these two helper functions:

  a) **timeAgo()** -- French relative time (no external library per project constraint):
  ```javascript
  function timeAgo(isoString) {
    if (!isoString) return null;
    var diff = Math.floor((Date.now() - new Date(isoString).getTime()) / 1000);
    if (diff < 60) return 'il y a ' + diff + ' s';
    if (diff < 3600) return 'il y a ' + Math.floor(diff / 60) + ' min';
    if (diff < 86400) return 'il y a ' + Math.floor(diff / 3600) + ' h';
    return 'il y a ' + Math.floor(diff / 86400) + ' j';
  }
  ```

  b) **renderStudentTable()** -- Builds table rows from stats response:
  ```javascript
  function renderStudentTable(students, totalFindings) {
    if (!students || !students.length) {
      document.getElementById('student-progress-tbody').innerHTML =
        '<tr><td colspan="5" style="text-align:center;color:#999;padding:1rem;">\u2014</td></tr>';
      document.getElementById('student-active-count').textContent = '0 actifs / 0 \u00e9tudiants';
      return;
    }

    var activeCount = students.filter(function(s) {
      if (!s.lastActiveAt) return false;
      return (Date.now() - new Date(s.lastActiveAt).getTime()) < 5 * 60 * 1000;
    }).length;

    document.getElementById('student-active-count').textContent =
      activeCount + ' actifs / ' + students.length + ' \u00e9tudiants';

    var tbody = document.getElementById('student-progress-tbody');
    tbody.innerHTML = students.map(function(s) {
      var pct = totalFindings ? Math.round(s.submitted / totalFindings * 100) : 0;
      var name = s.username.replace('_student', '');
      var ago = timeAgo(s.lastActiveAt);

      // Three-state badge: Actif (green), Inactif (amber), Pas commence (gray)
      var statusClass, statusLabel;
      if (s.submitted === 0 && !s.lastActiveAt) {
        statusClass = 'status-notstarted';
        statusLabel = 'Pas commenc\u00e9';
      } else if (s.lastActiveAt && (Date.now() - new Date(s.lastActiveAt).getTime()) < 5 * 60 * 1000) {
        statusClass = 'status-active';
        statusLabel = 'Actif';
      } else {
        statusClass = 'status-inactive';
        statusLabel = 'Inactif';
      }

      return '<tr>' +
        '<td><a href="/sca/student/' + s.id + '" style="color:#002855;font-weight:600;">' + name + '</a></td>' +
        '<td style="min-width:140px;">' +
          '<div class="progress-bar-wrap"><div class="progress-bar-fill" style="width:' + pct + '%"></div></div>' +
          '<div style="font-size:0.8rem;color:#666;margin-top:2px;">' + s.submitted + '/' + totalFindings + ' soumis</div>' +
        '</td>' +
        '<td style="font-size:0.85rem;color:#666;">' + (ago || '\u2014') + '</td>' +
        '<td style="font-size:0.85rem;">' + (s.currentFindingTitle || '\u2014') + '</td>' +
        '<td><span class="badge-sm ' + statusClass + '">' + statusLabel + '</span></td>' +
      '</tr>';
    }).join('');
  }
  ```
