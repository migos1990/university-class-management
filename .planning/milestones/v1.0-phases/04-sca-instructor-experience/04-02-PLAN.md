**B. Replace the `<script>` block** at the bottom of the file. The current script has the `importToVM` function with hardcoded English strings. Replace the entire `<script>...</script>` block with:

```ejs
<script>
// --- French constants (baked by EJS at render time) ---
const MSG_CONFIRM_IMPORT = '<%= t("sca.instructor.confirmImport") %>';
const MSG_IMPORTING = '<%= t("sca.instructor.importing") %>';
const MSG_IN_VM = '<%= t("sca.instructor.inVM") %>';
const MSG_PUSH_VM = '<%= t("sca.instructor.pushToVM") %>';
const MSG_NETWORK_ERROR = '<%= t("sca.common.networkError") %>';
const MSG_IMPORT_FAILED = '<%= t("sca.instructor.importFailed") %>';

// --- VM Import ---
async function importToVM(findingId, btn) {
  if (!confirm(MSG_CONFIRM_IMPORT)) return;
  btn.disabled = true;
  btn.textContent = MSG_IMPORTING;
  try {
    const res = await fetch('/sca/import-to-vm/' + findingId, { method: 'POST' });
    const data = await res.json();
    if (data.success) {
      btn.textContent = MSG_IN_VM;
      btn.className = 'action-btn btn-imported';
    } else {
      btn.disabled = false;
      btn.textContent = MSG_PUSH_VM;
      alert(data.error || MSG_IMPORT_FAILED);
    }
  } catch (e) {
    btn.disabled = false;
    btn.textContent = MSG_PUSH_VM;
    alert(MSG_NETWORK_ERROR);
  }
}

// --- Stats polling (30s) ---
async function refreshStats() {
  try {
    const res = await fetch('/sca/stats');
    const data = await res.json();
    document.getElementById('stat-started').textContent = data.studentsStarted + '/' + data.totalStudents;
    document.getElementById('stat-completion').textContent = data.avgCompletion + '%';
    document.getElementById('stat-pace').textContent = data.pace > 0
      ? data.pace + ' soumissions / 5 min'
      : '0';
    document.getElementById('stats-timestamp').textContent =
      'Mis \u00e0 jour : ' + new Date().toLocaleTimeString('fr-CA', { hour12: false });
  } catch (e) {
    // Silently fail -- stats will refresh on next interval
  }
}
refreshStats(); // Initial load on page open
setInterval(refreshStats, 30000); // Refresh every 30 seconds
</script>
```

Note: The `\u00e0` in the script produces the "a" in "Mis a jour" (with accent). This avoids encoding issues in the EJS template.
